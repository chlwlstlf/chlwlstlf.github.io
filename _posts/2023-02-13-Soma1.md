from itertools import combinations
import sys
input = sys.stdin.readline

N = int(input())
U = [int(input()) for _ in range(N)]

U.sort(reverse=True)

def cal(n):
  for i in range(n):
    for j in range(n):
      if U[j] > U[i]:
        continue
      for k in range(n):
        if U[j]+U[k] > U[i]:
          continue
        for l in range(n):
          if U[j]+U[k]+U[l] > U[i]:
            continue
          if U[j]+U[k]+U[l] == U[i]:
            return U[i]

print(cal(N))

import heapq
import sys

T = int(input())
for i in range(T):
  k = int(input())
  Q_min = []
  Q_max = []
  for i in range(k):
    o, n = map(str, input().rstrip().split())
    n = int(n)
    if o == 'I':
      heapq.heappush(Q_min, n)
      heapq.heappush(Q_max, -n)
    else:
      if len(Q_min) > 0:
        if n == -1:
          Q_max.remove(-heapq.heappop(Q_min))
          heapq.heapify(Q_max)
        else:
          Q_min.remove(-heapq.heappop(Q_max))
          heapq.heapify(Q_min)
  if len(Q_min) == 0:
    print('EMPTY')
  else:
    Q_min = list(Q_min)
    print(-Q_max[0], Q_min[0])

  if cnt == M:
    print(' '.join(map(str, arr)))
    return
  for i in range(N):
    if len(arr) == 0 or arr[-1] <= U[i]:
      arr.append(U[i])
      dfs(cnt+1, arr)
      arr.pop()
