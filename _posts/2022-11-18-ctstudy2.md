import sys
from itertools import combinations
input = sys.stdin.readline

N = int(input())
T = [list(map(int, input().split())) for _ in range(N)]

result = sys.maxsize

  start = 0
  link = 0
  l = list(set(idx) - set(s))
  for i, j in list(combinations(s, 2)) :
    start += T[i][j] + T[j][i]

    link += T[i][j] + T[j][i]
  result = min(result, abs(start-link))
